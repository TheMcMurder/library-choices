    **2. Replace the single-line change listener (line 117 `form.addEventListener('change', updateResult);`) with a function that calls updateResult AND dispatches analytics:**
    ```javascript
    form.addEventListener('change', function (e) {
      updateResult();
      if (e.target.matches('input[name="staffing"]')) {
        var level = window.LIBRARY_DATA.staffingLevels.find(function (l) { return l.id === e.target.value; });
        if (level) trackStaffingSelected(e.target.value, level.label);
      }
      if (e.target.id === 'collections-digital') {
        var idx = parseInt(e.target.value, 10);
        var opts = window.LIBRARY_DATA.collectionsDigital.options;
        if (opts[idx]) trackDigitalSliderChanged(opts[idx].value);
      }
      if (e.target.id === 'collections-physical') {
        var idx = parseInt(e.target.value, 10);
        var opts = window.LIBRARY_DATA.collectionsPhysical.options;
        if (opts[idx]) trackPhysicalSliderChanged(opts[idx].value);
      }
      if (e.target.matches('input[name="cities"]')) {
        var city = window.LIBRARY_DATA.cities.find(function (c) { return c.id === e.target.value; });
        if (city) trackCityToggled(e.target.value, city.label, e.target.checked);
      }
    });
    ```

    Per D-07 and D-08: events fire on `change` only (NOT `input`). City label comes from `window.LIBRARY_DATA.cities` (NOT DOM scraping — per Research pitfall 4). Slider analytics use `e.target.id` check (per Research pitfall 3 — tick button dispatch fires `change` from the slider element itself, so this is correct).

    **CRITICAL:** Do NOT add any analytics calls to the `form.addEventListener('input', ...)` listener (line 120). That is for live UI updates during drag.

    **3. Add trackBreakdownOpened to the existing toggleBtn click handler (currently at line 102-107).** Add `if (isHidden) trackBreakdownOpened();` inside the click handler, after `breakdownDetail.hidden = !isHidden;`. This fires only on OPEN (when isHidden was true before toggle), not on close.

    The click handler should become:
    ```javascript
    toggleBtn.addEventListener('click', function (e) {
      e.stopPropagation();
      var isHidden = breakdownDetail.hidden;
      breakdownDetail.hidden = !isHidden;
      toggleBtn.setAttribute('aria-label', isHidden ? 'Hide cost breakdown' : 'Show cost breakdown');
      if (isHidden) trackBreakdownOpened();
    });
    ```

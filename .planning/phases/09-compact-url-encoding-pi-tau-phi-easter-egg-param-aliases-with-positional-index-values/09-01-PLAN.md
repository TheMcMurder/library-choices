    var params = new URLSearchParams();
    // staffing: params.set('staffing', staffing.value)
    // collections: params.set('collections', collections.value)
    // cities: params.set('cities', cities.join(','))
    var qs = params.toString();
    history.replaceState(null, '', qs ? '?' + qs : location.pathname);
  }

  function restoreFromUrl() {
    var params = new URLSearchParams(location.search);
    if (!params.toString()) return;
    // staffing: params.get('staffing') -> validate against data.staffingLevels[].id
    // collections: params.get('collections') -> validate against data.collections.options[].value
    // cities: params.get('cities') -> split(','), validate against data.cities[].id
  }

  restoreFromUrl();
  form.dispatchEvent(new Event('change'));
  form.dispatchEvent(new Event('input'));
  form.addEventListener('change', encodeUrl);
}());
